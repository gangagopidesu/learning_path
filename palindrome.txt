# palindrome of a number

def palindrome(n):
    temp = n
    sum=0
    while temp!=0:
        remainder = temp % 10
        sum = sum * 10 + remainder
        temp = temp // 10
    if sum == n:
        return '{} is a palindrome number'.format(n)
    else:
        return '{} is not a palindrome number'.format(n)


input =int(input())
print(palindrome(input))
