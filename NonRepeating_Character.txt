# first NonReapeating Character of a string

def nonRepeatin(string):
    dictionary={}
    for i in string:
        if i not in  dictionary:
            dictionary[i]=1
        else:
            dictionary[i]+=1
    temp=False
    for key,value in dictionary.items():
        if value==1:
            temp=True
            return key
            break
    if temp==False:
        return -1
input=input()
print(nonRepeatin(input))
